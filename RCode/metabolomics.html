Network information

Network information is needed for NetGSA, PathNet and DEGraph, which is extracted from KEGG metabolic interactions. Since PathNet assess the significance of each pathway independently from the others, we only include binary interactions between metabolites from the same pathway. Interactions that show up in multiple pathways are recorded multiple times. The network for DEGraph is stored in g, whereas the network for NetGSA is stored in A.

Enrichment analysis

The following code explains in details how the data are simulated, and enrichment results collected. In brief, we randomly select 20 pathways from which 10% of metabolites (DC) are set to be affected variables. To generate simulated data, we first standardize the original data such that each row (metabolite) has mean zero and unit variance. A non-zero mean signal is then added to each affected metabolite to create truly enriched pathways. Lastly, we evaluate the empirical power for each pathway using different methods. All powers are averaged over 200 replications.

nCluster <- 5 # please set the number of clusters for parallel computing
muvals <- 0.5
detect_call <- 0.1
alpha <- 0.05
nreps <- 200
ncond <- 2
p <- nrow(dataset$dat)
npath <- nrow(kegg_pathways)
npath_selected <- 20

# For the selected pathways, assign which variables to affect uniformly.
setList <- lapply(1:nrow(kegg_pathways), function(i) which(kegg_pathways[i,]==1))
names(setList) <- rownames(kegg_pathways)
pathway_sig_id <- sample(1:npath, npath_selected)
isEnrichment <- rep(0, npath)
isEnrichment[pathway_sig_id] <- 1  
genesets <- lapply(1:npath, 
                   function(i) if (i %in% pathway_sig_id){
                     sample(names(setList[[i]]), ceiling(detect_call*length(setList[[i]])))
                     })
variables2affect <- unique(Reduce("c", genesets[pathway_sig_id]))

mu <- vector("list", ncond)
mu[[1]] <- rep(0, p)
mu[[2]] <- muvals * (match(dataset$metab_info$CID, variables2affect, nomatch = 0)>0)

#********************************
#****   Main iterations   *******
#********************************
dat <- vector("list", ncond)
dat[[1]] <- dataset$dat[,(dataset$sample_info$Label=="Diabetic")]    ## p by n1
dat[[2]] <- dataset$dat[,(dataset$sample_info$Label=="Non-Diabetic")]## p by n2
n <- sapply(dat, ncol)
dat <- lapply(dat, function(d) t(scale(t(d))))
dat_combined <- cbind(dat[[1]], dat[[2]])

my.iterations <- function(iters){
  set.seed(iters)
  sigInd <- matrix(0, npath, 12)
  colnames(sigInd) <- c("PathNet-nom", "PathNet-fdr","PathNet-fwer",
                        "camera-nom",  "camera-fdr", "camera-fwer",
                        "DEGraph-nom", "DEGraph-fdr","DEGraph-fwer",
                        "NetGSA-nom", "NetGSA-fdr","NetGSA-fwer")

  # Add the following four lines if shuffling the samples. 
  # ind.1 <- sample(ncol(dataset$dat), n[1])
  # dat <- vector("list", ncond) 
  # dat[[1]] <- dat_combined[,ind.1]
  # dat[[2]] <- dat_combined[,-ind.1]

  for (i in 1:ncond) {
    epsilonVar <- matrix(rnorm(p * n[i]), p, n[i])
    dat[[i]] <- dat[[i]] + matrix(rep(mu[[i]], n[i]), p, n[i]) + 0.5*epsilonVar
  }
  
  ##******************************
  ## (1) *******PathNet*******
  ##******************************
  ## Two-sample t-test to first determine the DE genes;
  pvals4metabs <- sapply(1:p, function(i) t.test(dat[[2]][i,], dat[[1]][i,], var.equal = FALSE)$p.value)
  dat4PathNet <- cbind(dataset$metab_info$Number,-log10(pvals4metabs))
  colnames(dat4PathNet) <- c("CID", "pvals")

  resPathNet <- PathNet(Enrichment_Analysis = TRUE, Contextual_Analysis = FALSE, 
                        DirectEvidence_info = dat4PathNet, Column_DirectEvidence = 2, 
                        Adjacency = A4PathNet, pathway = pathways4PathNet, 
                        n_perm = 100, threshold = alpha, use_sig_pathways  = FALSE)
  
  dict.pathnet <- data.frame("full.name" = sort(rownames(kegg_pathways)), 'short.name' = sort(as.character(resPathNet$enrichment_results$Name)))
  pathnet.tempName <- dict.pathnet$full.name[match(as.character(resPathNet$enrichment_results$Name),as.character(dict.pathnet$short.name))]
  resPathNet$enrichment_results$Name <- as.character(pathnet.tempName)
  resPathNet$enrichment_results <- resPathNet$enrichment_results[match(rownames(kegg_pathways),resPathNet$enrichment_results$Name),]

  sigInd[,1] <- (resPathNet$enrichment_results$p_PathNet<alpha)
  sigInd[,2] <- (p.adjust(resPathNet$enrichment_results$p_PathNet, "BH")<alpha)
  sigInd[,3] <- (p.adjust(resPathNet$enrichment_results$p_PathNet, "holm")<alpha)
  
  ##*******************************  
  ## (2) *******camera*******
  ##*******************************
  classT1D <- 2-as.numeric(dataset$sample_info$Label)
  names(classT1D) <- paste0(dataset$sample_info$Label,dataset$sample_info$Number)
  T1Dprofile <- cbind(dat[[1]],dat[[2]])
  
  design <- cbind(Intercept = 1, Group = classT1D)
  
  resCamera <- limma::camera(T1Dprofile, setList, design) 
  resCamera <- resCamera[match(rownames(kegg_pathways),rownames(resCamera)),]
  sigInd[,4] <- (resCamera$PValue < alpha)
  sigInd[,5] <- (p.adjust(resCamera$PValue, 'BH')<alpha)
  sigInd[,6] <- (p.adjust(resCamera$PValue, 'holm')<alpha)
  
  ##*******************************  
  ## (3) *******DEGraph*******
  ##*******************************
  res.DEGraph <- rep(NA, npath)
  ## Run NetGSA on each pathway
  for (loop_path in 1:npath){
    currentDat <- T1Dprofile[(rownames(T1Dprofile)%in% names(setList[[loop_path]])),]
    g_c <- induced_subgraph(g, names(setList[[loop_path]]))
    X1 <- t(currentDat[, which(dataset$sample_info$Label=='Diabetic')])
    X2 <- t(currentDat[, which(dataset$sample_info$Label=='Non-Diabetic')])
    res.DEGraph[loop_path] <- graph.T2.test(X1, X2, G=igraph.to.graphNEL(g_c))$p.value
  }
  sigInd[,7] <- (res.DEGraph<alpha)
  sigInd[,8] <- (p.adjust(res.DEGraph, "BH")<alpha)
  sigInd[,9] <- (p.adjust(res.DEGraph, "holm")<alpha)
  
  ##*******************************
  ## (4) *******NetGSA*******
  ##*******************************
  ## Estimate the partial correlation matrices
  wAdj <- vector("list", ncond)
  for (k in 1:ncond) {
    lambda_vec = seq(1, 5, 0.1) * sqrt(log(p)/n[k])
    bic_tune <- bic.netEst.undir(x=dat[[k]], one=A, lambda = lambda_vec, rho=0.1 * sqrt(log(p)/n[k]), eta=0.1)
    lambda_opt <- (9+which.min(bic_tune$BIC)) * sqrt(log(p)/n[k]) /10
    fit <- netEst.undir(dat[[k]], one=A, lambda = lambda_opt, rho=0.1 * sqrt(log(p)/n[k]), eta=0.1)
    wAdj[[k]] <- fit$Adj
  }

  out <- NetGSA(A = wAdj, x = T1Dprofile, group = c(rep(1, n[1]), rep(2, n[2])), pathways = kegg_pathways, lklMethod = "REML")
  # Output netgsa results
  sigInd[,10] <- (out$results$pval < alpha)
  sigInd[,11] <- (p.adjust(out$results$pval, "BH") < alpha)
  sigInd[,12] <- (p.adjust(out$results$pval, 'holm')<alpha)

  return(sigInd)
}


cat("In the main loop ... \n")
cl <- makeCluster(nCluster)
registerDoParallel(cl)

res <- foreach(iter = 1:nreps,#.combine='+',
               .packages = c("glasso","glmnet","graphite","igraph","KEGGgraph",
                             "PathNet","limma","DEGraph","org.Hs.eg.db"), .verbose = T) %dopar% {
                               tryCatch({my.iterations(iter)}, error=function(e) NULL)
                             }

stopCluster(cl)

# The average power
sigInd <- Reduce("+", res)/length(res)
